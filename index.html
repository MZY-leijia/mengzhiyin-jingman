CREATE TABLE users ( username TEXT PRIMARY KEY, password TEXT, coins INT DEFAULT 0, membership_expire BIGINT DEFAULT 0, avatar TEXT DEFAULT '😊', bio TEXT DEFAULT '追番爱好者', role TEXT DEFAULT 'user', sign_streak INT DEFAULT 0, last_sign_date TEXT DEFAULT '' ); CREATE TABLE comments ( id SERIAL PRIMARY KEY, anime_id INT, username TEXT, content TEXT, created_at BIGINT ); CREATE TABLE favorites ( username TEXT, anime_id INT, PRIMARY KEY (username, anime_id) ); CREATE TABLE watch_history ( username TEXT, anime_id INT, title TEXT, cover TEXT, episode TEXT, timestamp BIGINT, PRIMARY KEY (username, anime_id) ); CREATE TABLE gift_codes ( code TEXT PRIMARY KEY, value INT, used BOOLEAN DEFAULT FALSE, created_at BIGINT );
